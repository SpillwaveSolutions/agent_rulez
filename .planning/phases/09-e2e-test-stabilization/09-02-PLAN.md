```yaml
# Cross-Platform E2E Tests for RuleZ Core
#
# Runs E2E tests (rulez/tests/e2e_*.rs) on ubuntu, macOS, and Windows.
# Includes binary artifact validation to catch stale cached binaries.
#
# For Playwright UI tests, see e2e.yml (separate workflow).

name: E2E Tests - Cross-Platform

on:
  push:
    branches: [main, develop, "feature/**", "fix/**"]
    paths:
      - 'rulez/**'
      - '.github/workflows/e2e-matrix.yml'
  pull_request:
    branches: [main, develop]
    paths:
      - 'rulez/**'
      - '.github/workflows/e2e-matrix.yml'
  workflow_dispatch:

env:
  CARGO_TERM_COLOR: always
  RUST_BACKTRACE: 1

jobs:
  e2e-matrix:
    name: E2E (${{ matrix.os }})
    strategy:
      fail-fast: false
      matrix:
        os: [ubuntu-latest, macos-latest, windows-latest]

    runs-on: ${{ matrix.os }}
    timeout-minutes: 20

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Setup Rust toolchain
        uses: dtolnay/rust-toolchain@stable

      - name: Cache dependencies
        uses: Swatinem/rust-cache@v2

      - name: Build binary
        run: cargo build --bin rulez

      - name: Validate binary artifact (Unix)
        if: runner.os != 'Windows'
        run: |
          BINARY_PATH="target/debug/rulez"

          # Check binary exists
          if [ ! -f "$BINARY_PATH" ]; then
            echo "::error::Binary not found at $BINARY_PATH"
            exit 1
          fi

          # Verify it reports the correct name
          ./$BINARY_PATH --version
          ./$BINARY_PATH --version | grep -q "rulez" || {
            echo "::error::Binary version check failed — may be stale 'cch' binary"
            exit 1
          }

          # Check for stale binary from old name
          if [ -f "target/debug/cch" ]; then
            echo "::warning::Stale 'cch' binary found in target/debug/ — consider cleaning cache"
          fi

          echo "Binary validation passed: $BINARY_PATH"

      - name: Validate binary artifact (Windows)
        if: runner.os == 'Windows'
        shell: pwsh
        run: |
          $binaryPath = "target\debug\rulez.exe"

          if (-not (Test-Path $binaryPath)) {
            Write-Error "Binary not found at $binaryPath"
            exit 1
          }

          $version = & ".\$binaryPath" --version 2>&1
          if ($version -notmatch "rulez") {
            Write-Error "Binary version check failed — may be stale 'cch' binary"
            exit 1
          }

          # Check for stale binary
          if (Test-Path "target\debug\cch.exe") {
            Write-Warning "Stale 'cch.exe' binary found in target\debug\"
          }

          Write-Host "Binary validation passed: $binaryPath"

      - name: Run all tests
        run: cargo test --tests --all-features --workspace

      - name: Upload test evidence
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: test-evidence-${{ matrix.os }}
          path: rulez/target/test-evidence/
          retention-days: 7
          if-no-files-found: ignore

      - name: Upload failure logs
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: failure-logs-${{ matrix.os }}
          path: |
            rulez/target/test-evidence/
          retention-days: 14
          if-no-files-found: ignore
```
